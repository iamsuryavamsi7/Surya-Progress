	
	Git Commands :- 
	============

	1)  git --version
	2)  git config --global user.name
	3)  git config --global user.email
	4)  git init
	5)  git add "File Name" (or) git add . (Current Directory all files)
	6)  git rm --cached "File Name" (or) git rm -r --cached . (All Files in staged area)
	7)  git commit -m "Type the message here"
	8)  git log (or) git log --oneline
	9)  git status
	10) git commit --amend -m "Type the new Commit Message here"
	11) git branch "Branch Name"
	12) git checkout "Branch Name"
	13) git checkout -b "Branch Name"
	14) git branch -d "Branch Name"
	15) git branch -D "Branch Name"
	16) git push --set-upstream origin main (or) git push 
	17) git pull --set-upstream origin main (or) git pull
